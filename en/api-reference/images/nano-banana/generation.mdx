## Body

<ParamField body="model" type="string" required>
  Model Name

  Fixed value: `"nano_banana"`
</ParamField>

<ParamField body="params" type="object" required>
  Model Parameters Object

  <Expandable title="params Object Properties">
    <ParamField body="prompt" type="string" required>
      Text description for image generation

      No limits
    </ParamField>

    <ParamField body="image_url" type="array">
      List of reference image URLs for image-to-image generation

      * Up to 9 images
      * Supports publicly accessible URLs
    </ParamField>

    <ParamField body="aspect_ratio" type="string" default="auto">
      Aspect ratio for image generation

      Supported aspect ratios:

      * `auto` - Automatic (default)
      * `1:1` - Square
      * `3:2` - Horizontal 3:2
      * `2:3` - Vertical 2:3
      * `4:3` - Horizontal 4:3
      * `3:4` - Vertical 3:4
      * `5:4` - Horizontal 5:4
      * `4:5` - Vertical 4:5
      * `16:9` - Horizontal widescreen
      * `9:16` - Vertical portrait
      * `21:9` - Ultra widescreen
    </ParamField>

    <ParamField body="format" type="string" default="png">
      Image output format

      Supported formats:

      * `png` - PNG format (default)
      * `jpeg` - JPEG format
    </ParamField>
  </Expandable>
</ParamField>

  Initiator Task ID

  User-defined task identifier, required
</ParamField>

<ParamField body="callback_url" type="string">

          curl_easy_cleanup(curl);

                      return;
                  }
                  NSString *result = [[NSString alloc] initWithData:data
                                                          encoding:NSUTF8StringEncoding];
                  NSLog(@"%@", result);
              }];

          [task resume];
          [[NSRunLoop mainRunLoop] run];
      }
      return 0;
  }
